    `
	function Myboolean(a){
    if(typeof a =='number'){
        if(( a !==0||a!==NaN)){
        return true
        }
    }
    if(typeof a =='boolean'){
        if(a===true){
            return true
        }
        }
    if(typeof a =='object'){
        if(a!==null){
            return true
        }
    }
    if(typeof a =='string'){
        if(a.length!==0){
            return true
        }
    }
    return false
}
	`
